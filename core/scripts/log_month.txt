                meter_name,
                ROUND(SUM(production),2) as 'production'
            FROM
                `tbl_hourly_prod`
            WHERE
                MONTH(DATETIME) = 4 AND YEAR(DATETIME) = 2025 AND  meter_id = 100
            GROUP BY
                DATE(DATETIME)

                SELECT
                    date AS 'date',
                    meter_name,
                    ROUND(active_power, 2) AS 'active_power'
                FROM
                    `plant_active_power`
                WHERE
                    MONTH(date) = 4 AND YEAR(date) = 2025 AND meter_id = 100

            SELECT
                DATE(date) AS 'date',
                ROUND(sum(insolation),2) as 'insolation'
            FROM
                `plant_irradiance`
            WHERE
                MONTH(date) = 4 AND YEAR(date) = 2025
            GROUP BY DATE(date)
